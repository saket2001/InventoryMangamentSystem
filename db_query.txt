create table product(
    userid varchar(20) not null,
    productid int not null primary key,
    product_name varchar(40) not null,
    product_category varchar(30) not null,
    product_price varchar(20) not null,
    product_maxlimit int not null,
    product_minlimit int not null,
    product_instock int,
    supplier_id int not null,
    foreign key (userid) references user(username)
)

create table user(
    username varchar(30) primary key not null,
    password varchar(30) not null,
    name varchar(50) not null,
    dob date not null,
    contact_no numeric(10) not null
)


create table supplier(
 userid varchar(20) not null,
 supplier_id int primary key not null ,
 supplier_name varchar(50) not null,
 supplier_type varchar(30) not null,
 supplier_contactno numeric,
 foreign key (userid) references user(username)
);

create table billing_record(
	userid varchar(30) not null,
	bill_id varchar(10) not null primary key,
    customer_name varchar(30) not null,
    bill_date varchar(20) not null,
    bill_total double not null,
    foreign key (userid) references user(username)
)
